function startListening() { const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition; if (!SpeechRecognition) { alert("Speech recognition not supported in this browser"); return; } const recognition = new SpeechRecognition(); recognition.lang = "en-US"; recognition.start(); recognition.onresult = function(event) { let text = event.results[0][0].transcript; document.getElementById("spoken").innerText = text; let math = convert(text); try { let answer = eval(math); document.getElementById("result").innerText = "= " + answer; speak("The answer is " + answer); } catch { document.getElementById("result").innerText = "Error"; } }; }
